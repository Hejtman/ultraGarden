from decimal import Decimal


def n_angle_side():
    side = Decimal(0.5)
    while True:
        yield side
        side = ((1+side)**Decimal(0.5) - (1-side)**Decimal(0.5))/2

for n, s in zip(range(26), n_angle_side()):
    print "{}. {}-angle side: {} circumference: {}".format(n, 6*2**n, s, 6*2**n*s)
